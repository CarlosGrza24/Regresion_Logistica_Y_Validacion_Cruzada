import pandas as pd
import numpy as np

df = pd.read_csv("CA2022_Cultivos_Superficie_Produccion.csv", encoding="latin-1")

# Eliminar columnas no necesarias
df = df.drop(columns=["NOM_MUN", "NOMBRE", "ENTIDAD"], errors="ignore")
target = "Total UP agropecuaria activas"
df = df.dropna(subset=[target])

#  Outliers (Tukey) 
Q1 = df[target].quantile(0.25)
Q3 = df[target].quantile(0.75)
IQR = Q3 - Q1
lim_inf = Q1 - 1.5 * IQR
lim_sup = Q3 + 1.5 * IQR
df_tukey = df[(df[target] >= lim_inf) & (df[target] <= lim_sup)].copy()

# Colinealidad
base = df_tukey if 'df_tukey' in globals() else df
y = base[target].copy()
cols_excluir = {target, "UP Agricolas", "y_winz", "y_log", "es_outlier"} & set(base.columns)
X = (
    base.drop(columns=list(cols_excluir), errors="ignore")
        .select_dtypes(include="number")
        .copy()
)
#  Eliminar colinealidad alta 
corr_matrix = X.corr().abs()
upper = corr_matrix.where(np.triu(np.ones(corr_matrix.shape), k=1).astype(bool))
to_drop = [column for column in upper.columns if any(upper[column] > 0.85)]
X = X.drop(columns=to_drop, errors="ignore")

# Base final 
df_final = X.copy()
df_final[target] = y

Esto divide el conjunto de entrenamiento en 5 folds que conservan la proporcin 0/1; en cada iteracin se entrena en 4 folds y se evala en el restante, repitiendo el proceso 5 veces.

Se emple un Pipeline con StandardScaler este evita data leakage y mejora la estabilidad numrica del modelo. Reportamos accuracy, recall, precision, F1 y ROC-AUC.

  • Precision: 0.747 0.036 de las predicciones alta produccin, 75% realmente lo son (25% falsos positivos).

  • F1: 0.649 0.025 equilibrio entre recall y precision; desempeo consistente.

  • ROC-AUC: 0.756 0.029 discriminacin moderada-buena y estable entre folds.

  • Accuracy: 0.689 0.023 69% de aciertos totales.

El modelo tiende a ser conservador (alta precisin frente a un recall moderado): cuando predice alta produccin suele acertar, aunque deja pasar algunos positivos. La baja desviacin estndar en todas las mtricas indica estabilidad del rendimiento.

from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import StratifiedKFold, cross_validate

pipe = Pipeline(steps=[
    ("scaler", StandardScaler()),
    ("clf", LogisticRegression(solver="lbfgs", max_iter=1000))  
])
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
scoring = {
    "accuracy":  "accuracy",
    "recall":    "recall",
    "precision": "precision",
    "f1":        "f1",
    "roc_auc":   "roc_auc"
}
cv_res = cross_validate(
    pipe, X_train, y_train,

)

import matplotlib.pyplot as plt
from sklearn.metrics import (
    confusion_matrix, ConfusionMatrixDisplay,
    accuracy_score, recall_score, precision_score, f1_score
)

pipe.fit(X_train, y_train)
y_prob = pipe.predict_proba(X_test)[:, 1]
print("Primeras 10 probabilidades:", np.round(y_prob[:10], 3))

def eval_thresholds(y_true, y_prob, thresholds=(0.5, 0.3, 0.7)):
    filas, cms = [], {}
    for t in thresholds:
        y_pred = (y_prob >= t).astype(int)
        cm = confusion_matrix(y_true, y_pred, labels=[0, 1])
        tn, fp, fn, tp = cm.ravel()

        acc  = accuracy_score(y_true, y_pred)
        sens = recall_score(y_true, y_pred)                 
        esp  = tn / (tn + fp) if (tn + fp) else np.nan      
        prec = precision_score(y_true, y_pred, zero_division=0)
        f1   = f1_score(y_true, y_pred)

        filas.append({
            "umbral": t, "accuracy": acc, "sensibilidad": sens,
            "especificidad": esp, "precision": prec, "f1": f1,
            "TN": tn, "FP": fp, "FN": fn, "TP": tp
        })
        cms[t] = cm

    tabla = pd.DataFrame(filas).sort_values("umbral")
    return tabla, cms

tabla_4, cms_4 = eval_thresholds(y_test, y_prob, thresholds=(0.5, 0.3, 0.7))

6- En este punto interpretamos la regresin logstica para entender cmo cada variable se asocia con la probabilidad de alta produccin (clase 1). Trabajamos con el modelo final (Pipeline = StandardScaler + LogisticRegression) y extraemos sus coeficientes estandarizados () junto con sus odds ratios (OR = e^). Previamente controlamos la colinealidad alta para evitar inestabilidad en los coeficientes y hacer la lectura ms confiable.

  • Signo de : positivo aumenta la probabilidad de alta produccin; negativo la reduce.

  • Odds ratio (OR): es e^ (exponencial de ). Convierte el coeficiente a un cambio multiplicativo en las odds (p/(1p)). OR > 1 favorece la clase 1; OR < 1 la desfavorece.
